  sync: function(method, model, options){
    var resp;
    var self = this;

    switch (method) {
      case "read":
        this.syncStore.get(null, function(items){
          if(chrome.extension.lastError){
            options.error(chrome.extension.lastError.message);
          }else{
            self.set(items);
            options.success();
          }
        });
        break;
      case "create":
      case "update":
        this.syncStore.set(model.attributes, function(){
          if(chrome.extension.lastError){
            options.error(chrome.extension.lastError.message);
          }else{
            options.success();
          }
        });
    }
  }
