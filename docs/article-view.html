  postRender: function(){
    $(".timeago").timeago();

  createArtilceView: function(article){
    return new App.ArticleView({
      model:      article,
      id:         "article-" + article.cid,
      className:  "news_item in_category_" + article.get("categoryEnglish").underscore()
    });
  }
});
